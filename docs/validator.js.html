import { calculateAge } from "./module.js";

/**
 * Vérifie qu'une personne est majeure (>= 18 ans) en utilisant `calculateAge`.
 *
 * @param {object} person - Objet personne.
 * @param {Date} person.birth - Date de naissance.
 * @throws {Error} - Si la personne a moins de 18 ans (ou si l'entrée est invalide via calculateAge).
 * @returns {number} - L'âge de la personne si elle a 18 ans ou plus.
 */
export function validateAdult(person) {
  const age = calculateAge(person);

  if (age < 18) {
    throw new Error("Pegi 18");
  }

  return age;
}

/**
 * @param {string} postalCode - code postal a 5 chiffre.
 * @throws {Error} - "Zip required" si manquant
 * @throws {Error} - "Zip code type error" si pas une string
 * @throws {Error} - "Zip code wrong length" si pas exactement 5 chiffres
 * @return {string} - Si le code postal est valide, retourne le code postal.
 */
export function validatePostalCode(postalCode) {
  if (postalCode === null || postalCode === undefined || postalCode === "") {
    throw new Error("Zip required");
  }
  if (typeof postalCode !== "string") {
    throw new Error("Zip code type error");
  }
  if (!/^\d{5}$/.test(postalCode)) {
    throw new Error("Zip code wrong length");
  }
  return postalCode;
}

/**
 * @param {string} email - adresse email à valider
 * @throws {Error} - "Email is required" si manquant
 * @throws {Error} - "Email must be a string" si pas une string
 * @throws {Error} - "Invalid email format" si format invalide
 * @return {string} - Si l'email est valide, retourne l'email.
 */

export function validateEmail(email) {
  if (email === null || email === undefined || email === "") {
    throw new Error("Email is required");
  }
  if (typeof email !== "string") {
    throw new Error("Email must be a string");
  }

  const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  if (!re.test(email)) {
    throw new Error("Invalid email format");
  }
  return email;
}

/**
 * @param {object} identity - objet avec les propriétés firstName et lastName
 * @throws {Error} - "Identity is required" si manquant
 * @throws {Error} - "Identity must be an object" si pas un objet valide
 * @throws {Error} - "firstName and lastName are required" si champs manquants
 * @throws {Error} - "firstName/lastName must be strings" si mauvais types
 * @throws {Error} - "HTML tags are not allowed in identity" si HTML détecté
 * @throws {Error} - "Invalid identity format" si caractères non autorisés
 * @return {object} - Si l'identité est valide, retourne l'objet identity.
 */
export function validateIdentity(identity) {
  if (identity === null || identity === undefined) {
    throw new Error("Identity is required");
  }
  if (typeof identity !== "object" || Array.isArray(identity)) {
    throw new Error("Identity must be an object");
  }

  const { firstName, lastName } = identity;

  if (!firstName || !lastName) {
    throw new Error("firstName and lastName are required");
  }
  if (typeof firstName !== "string" || typeof lastName !== "string") {
    throw new Error("firstName/lastName must be strings");
  }

  const hasHtmlTag = /<[^>]*>/.test(firstName) || /<[^>]*>/.test(lastName);
  if (hasHtmlTag) {
    throw new Error("HTML tags are not allowed in identity");
  }
  const nameRe = /^[A-Za-zÀ-ÖØ-öø-ÿ]+(?:[ '-][A-Za-zÀ-ÖØ-öø-ÿ]+)*$/;

  if (!nameRe.test(firstName) || !nameRe.test(lastName)) {
    throw new Error("Invalid identity format");
  }

  return identity;
}
